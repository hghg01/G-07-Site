              int on_off = 0;
              int button = 0;
              
              void setup() 
              {
              pinMode (13, OUTPUT);          
              pinMode (12, OUTPUT);
              pinMode (12, INPUT);
              digitalWrite(13, LOW);
              digitalWrite(12, LOW);
              on_off = LOW;
              }
            
              void loop() 
              {
              button = digitalRead (12); 
              if (button == HIGH) {
                if (on_off == LOW) {
                digitalWrite(13, HIGH);
                on_off= HIGH;
                } else {
                  digitalWrite(13, LOW);
                  on_off = LOW;
                }
              }
              delay(10); // Delay a little bit to improve simulation
              }
